/* * PATCH สำหรับ inventory.html * ---------------------------------------------------------------- * แทนที่ 4 ฟังก์ชันนี้ทั้งหมดในไฟล์เดิม: * loadInventoryData(), saveAdjustment(), saveIngredient(), saveSupplier() * และเพิ่มฟังก์ชันใหม่ loadTransactions() ที่ต้องเรียกตอนสลับไป tab Transactions * (แก้ใน switchTab() ด้วย — ดูจุดที่ 5 ด้านล่าง) * * สรุปสิ่งที่เปลี่ยน: * - loadInventoryData() → เปลี่ยนจาก API.inventory() เป็น API.ingredients() * (ผูกกับ ingredients.current_stock ตัวเดียวกับที่ POS หักตอนขายจริง) * - saveAdjustment() → เรียก API.updateStock()/API.recordWastage() จริง * แทนการแก้แค่ array ใน browser แล้ว reload ข้อมูลจาก backend ใหม่ * - saveIngredient() → เรียก API.createIngredient() จริง * - saveSupplier() → เรียก API.createSupplier() จริง แล้ว reload * - loadTransactions() → ฟังก์ชันใหม่ ดึงประวัติจริงจาก API.stockTransactions() * ---------------------------------------------------------------- */ // ═══ 1) แทนที่ loadInventoryData() เดิมทั้งหมด ═══════════════════════════════ async function loadInventoryData() { try { const data = await API.ingredients(); if (data && data.length) { INVENTORY = data.map(i => ({ id: i.ingredient_id, name: i.name, sku: i.ingredient_id, category: i.category || 'other', brand: i.brand_id || 'SHARED', // จะเป็น SHARED เสมอจนกว่าจะเพิ่มคอลัมน์ brand_id ใน sheet ingredients unit: i.unit, stock: Number(i.current_stock || 0), min_stock: Number(i.min_stock || 0), max_stock: Number(i.max_stock || (Number(i.min_stock || 0) * 5)), unit_cost: Number(i.unit_cost || 0), supplier: i.supplier || '—', })); } else { INVENTORY = []; } } catch(e) { console.error('โหลดข้อมูล inventory ไม่สำเร็จ:', e); UI.toast('โหลดข้อมูล inventory ไม่สำเร็จ — ลองรีเฟรชหน้าใหม่', 'error'); INVENTORY = []; } renderInventory(); } // ═══ 2) แทนที่ saveAdjustment() เดิมทั้งหมด ═══════════════════════════════════ // ความหมาย adjType (ต้องตรงกับที่ backend adjustIngredientStock() กำหนดไว้): // receive : บวกเข้า (รับของเข้า) — ตัด unit_cost ไปสร้าง expense อัตโนมัติด้วยถ้ากรอกไว้ // deduct : ลบออกด้วยมือ // waste : ของเสีย/หมดอายุ // adjust : ตั้งค่าสัมบูรณ์ตามที่นับได้จริง (ไม่ใช่ delta แล้ว! ต่างจาก mock เดิม) async function saveAdjustment() { const ingId = document.getElementById('adj-ingredient').value; const qty = parseFloat(document.getElementById('adj-qty').value); const cost = parseFloat(document.getElementById('adj-cost').value) || undefined; const note = document.getElementById('adj-note').value; if (qty === undefined || isNaN(qty)) { alert('กรุณาใส่จำนวน'); return; } const item = INVENTORY.find(i => i.id === ingId); if (!item) return; const btn = document.querySelector('#adjust-modal .btn-primary'); UI.loading(btn, true); try { const payload = { ingredient_id: ingId, adjust_type: adjType, quantity: qty, unit_cost: cost, // ใช้เฉพาะตอน receive เพื่อสร้าง expense อัตโนมัติ note: note || '', }; const res = adjType === 'waste' ? await API.recordWastage(payload) : await API.updateStock(payload); if (!res || res.success !== true) { throw new Error((res && res.error) || 'บันทึกไม่สำเร็จ'); } UI.toast( adjType === 'receive' && cost ? `รับของเข้าเรียบร้อย + บันทึกรายจ่าย ฿${(qty*cost).toLocaleString()} ให้อัตโนมัติ` : 'บันทึกการปรับสต๊อกเรียบร้อย' ); closeModal('adjust-modal'); await loadInventoryData(); // โหลดสต๊อกล่าสุดจาก backend จริง แทนการแก้ array มือ } catch(e) { console.error(e); UI.toast('บันทึกไม่สำเร็จ: ' + e.message, 'error'); } finally { UI.loading(btn, false); } } // ═══ 3) แทนที่ saveIngredient() เดิมทั้งหมด ═══════════════════════════════════ async function saveIngredient() { const name = document.getElementById('add-name').value.trim(); if (!name) { alert('กรุณาใส่ชื่อวัตถุดิบ'); return; } const btn = document.querySelector('#add-modal .btn-primary'); UI.loading(btn, true); try { const res = await API.createIngredient({ name, category: document.getElementById('add-cat').value, unit: document.getElementById('add-unit').value, current_stock: parseFloat(document.getElementById('add-stock').value) || 0, min_stock: parseFloat(document.getElementById('add-min').value) || 0, unit_cost: parseFloat(document.getElementById('add-cost').value) || 0, brand_id: document.getElementById('add-brand').value, // เก็บไว้เผื่อ sheet เพิ่มคอลัมน์ brand_id ทีหลัง }); if (!res || res.success !== true) throw new Error((res && res.error) || 'บันทึกไม่สำเร็จ'); UI.toast('เพิ่มวัตถุดิบเรียบร้อย: ' + res.ingredient_id); closeModal('add-modal'); await loadInventoryData(); } catch(e) { console.error(e); UI.toast('เพิ่มวัตถุดิบไม่สำเร็จ: ' + e.message, 'error'); } finally { UI.loading(btn, false); } } // ═══ 4) แทนที่ saveSupplier() เดิมทั้งหมด ═════════════════════════════════════ async function saveSupplier() { const name = document.getElementById('sup-name').value.trim(); if (!name) { alert('กรุณาใส่ชื่อ supplier'); return; } const btn = document.querySelector('#supplier-modal .btn-primary'); UI.loading(btn, true); try { const res = await API.createSupplier({ company_name: name, contact_person: document.getElementById('sup-contact').value, phone: document.getElementById('sup-phone').value, payment_terms: document.getElementById('sup-terms').value, products_supplied: document.getElementById('sup-products').value, }); if (!res || res.success !== true) throw new Error((res && res.error) || 'บันทึกไม่สำเร็จ'); UI.toast('เพิ่ม Supplier เรียบร้อย'); closeModal('supplier-modal'); const data = await API.suppliers(); if (data && data.length) { SUPPLIERS = data.map(s => ({ id: s.supplier_id, name: s.company_name || s.name, contact: s.contact_person || '—', phone: s.phone || '—', terms: s.payment_terms || 'cod', last_order: s.last_order || '—', products: s.products_supplied || '—', })); } renderSuppliers(); } catch(e) { console.error(e); UI.toast('เพิ่ม Supplier ไม่สำเร็จ: ' + e.message, 'error'); } finally { UI.loading(btn, false); } } // ═══ 5) เพิ่มฟังก์ชันใหม่ loadTransactions() ═════════════════════════════════ // วางไว้ใกล้ๆ renderTransactions() เดิมก็ได้ async function loadTransactions() { const body = document.getElementById('tx-table-body'); body.innerHTML = `
⏳
กำลังโหลด...
`; try { const data = await API.stockTransactions(); TRANSACTIONS = (data || []).map(t => ({ id: t.id, ingredient_id: t.ingredient_id, ingredient_name: t.ingredient_name, type: t.type === 'waste' ? 'waste' : (t.type === 'receive' ? 'receive' : (t.type === 'adjust' ? 'adjust' : 'deduct')), qty: t.qty, date: t.date, note: t.note || '—', unit: t.unit, })); } catch(e) { console.error('โหลด transactions ไม่สำเร็จ:', e); TRANSACTIONS = []; } renderTransactions(); } // ─── แก้ switchTab() เดิม — เปลี่ยนบรรทัด: // if (tab === 'transactions') renderTransactions(); // เป็น: // if (tab === 'transactions') loadTransactions(); // ═══ 6) แทนที่ updateAdjPreview() เดิมทั้งหมด ═════════════════════════════════ // (ความหมายของ 'adjust' เปลี่ยนจาก delta เป็นค่าสัมบูรณ์แล้ว ต้องแก้ preview ให้ตรง // ไม่งั้นตัวเลขที่โชว์ก่อนกดบันทึกจะไม่ตรงกับที่ backend จะคำนวณจริง) function updateAdjPreview() { const ingId = document.getElementById('adj-ingredient')?.value; const qty = parseFloat(document.getElementById('adj-qty')?.value) || 0; const item = INVENTORY.find(i => i.id === ingId); if (!item) return; let after; if (adjType === 'receive') after = item.stock + Math.abs(qty); else if (adjType === 'deduct' || adjType === 'waste') after = Math.max(0, item.stock - Math.abs(qty)); else after = Math.max(0, qty); // adjust = ตั้งค่าสัมบูรณ์ตามที่นับได้จริง document.getElementById('adj-preview').textContent = `${item.stock.toLocaleString()} ${item.unit} → ${after.toLocaleString()} ${item.unit}`; }
